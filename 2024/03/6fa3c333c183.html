Jedis操作数据类型

测试String

    @Test
    public void demo1() {
        /**
         * @description:
         * 测试String
         * @params: []
         * @return: void
         */

        Jedis jedis = new Jedis("192.168.0.116", 6379);

        jedis.mset("str1", "v1", "str2", "v2");//设置多个字符串

        Set<String> keys = jedis.keys("*"); //得到所有Redis中的所有key
        System.out.println("得到Redis所有键：");
        keys.forEach(System.out::println);
        System.out.println("得到str1的值：" + jedis.get("str1"));
        
        jedis.close();
    }

测试List

    @Test
    public void demo2() {
        /**
         * @description:
         * 测试List
         * @params: []
         * @return: void
         */

        Jedis jedis = new Jedis("192.168.0.116", 6379);

        jedis.lpush("list1", "lucy", "John", "Marx");

        List<String> list = jedis.lrange("list1", 0, -1);
        list.forEach(System.out::println);// Marx John lucy
        
        jedis.close();

    }

测试Hash

    @Test
    public void demo3() {
        /**
         * @description:
         * 测试hash
         * @params: []
         * @return: void
         */

        Jedis jedis = new Jedis("192.168.0.116", 6379);

        jedis.hset("hash1", "field1", "value1");
        System.out.println(jedis.hget("hash1", "field1"));
        
        jedis.close();
    }

测试Set

    @Test
    public void demo4() {
        /**
         * @description:
         * 测试Set
         * @params: []
         * @return: void
         */

        Jedis jedis = new Jedis("192.168.0.116", 6379);

        jedis.sadd("set1", "apple", "banana", "orange");
        System.out.println(jedis.smembers("set1"));//[banana, apple, orange]
        
        jedis.close();

    }

测试Zset

    @Test
    public void demo5() {
        /**
         * @description:
         * 测试Zset
         * @params: []
         * @return: void
         */

        Jedis jedis = new Jedis("192.168.0.116", 6379);

        jedis.zadd("zset1", 100, "apple");
        jedis.zadd("zset1", 200, "banana");
        jedis.zadd("zset1", 300, "orange");
        Set<String> zset1 = jedis.zrange("zset1", 0, -1);
        System.out.println(zset1);//[apple, banana, orange]
        
        jedis.close();
    }
